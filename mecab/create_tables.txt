create table M_HINSHI
(HINSHI_NO Bigint NOT NULL,
 HINSHI_BASIC Varchar(200) NOT NULL,
 PARENT_HINSHI_NO Bigint NULL,
 HINSHI_LEVEL int NOT NULL,
 CREATED_AT DATETIME NOT NULL,
 UPDATED_AT DATETIME NOT NULL,
 DELETE_FLAG TINYINT DEFAULT '0' NOT NULL,
 PRIMARY KEY(HINSHI_NO)
) ENGINE=INNODB;

create table M_KATSUYO
(KATSUYO_NO Bigint NOT NULL,
 KATSUYO_BASIC Varchar(200) NOT NULL,
 PARENT_KATSUYO_NO Bigint NULL,
 KATSUYO_LEVEL int NOT NULL,
 CREATED_AT DATETIME NOT NULL,
 UPDATED_AT DATETIME NOT NULL,
 DELETE_FLAG TINYINT DEFAULT '0' NOT NULL,
 PRIMARY KEY(KATSUYO_NO)
) ENGINE=INNODB;

create table M_SEED
(SEED_NO Bigint NOT NULL AUTO_INCREMENT,
 HYOSO_TYPE Varchar(200) NOT NULL,
 LEFT_CONNECT_STATUS_NO int NOT NULL,
 RIGHT_CONNECT_STATUS_NO int NOT NULL,
 HINSHI Bigint NOT NULL,
 HINSHI_DETAIL_1 Bigint NULL,
 HINSHI_DETAIL_2 Bigint NULL,
 HINSHI_DETAIL_3 Bigint NULL,
 KATSUYO_KEI Varchar(200) NULL,
 KATSUYO_TYPE Varchar(200) NULL,
 YOMI Varchar(200) NOT NULL,
 HATSUON Varchar(200) NOT NULL,
 CREATED_AT DATETIME NOT NULL,
 UPDATED_AT DATETIME NOT NULL,
 DELETE_FLAG TINYINT DEFAULT '0' NOT NULL,
 PRIMARY KEY(SEED_NO),
 UNIQUE(HYOSO_TYPE, HINSHI)
) ENGINE=INNODB;
